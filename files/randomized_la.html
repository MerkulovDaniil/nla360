import numpy as np

def freivalds_check(A, B, C, k=2):
    """
    –ü—Ä–æ–≤–µ—Ä—è–µ—Ç —Ä–∞–≤–µ–Ω—Å—Ç–≤–æ A @ B = C
    —Å –ø–æ–º–æ—â—å—é –∞–ª–≥–æ—Ä–∏—Ç–º–∞ –§—Ä–µ–π–≤–∞–ª–¥—Å–∞ k —Ä–∞–∑.
    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç True (—Å–∫–æ—Ä–µ–µ –≤—Å–µ–≥–æ) –∏–ª–∏ False (—Ç–æ—á–Ω–æ).
    """
    n = A.shape[0]
    for _ in range(k):
        # –°–ª—É—á–∞–π–Ω—ã–π –≤–µ–∫—Ç–æ—Ä (—ç–ª–µ–º–µ–Ω—Ç—ã -1 –∏–ª–∏ 1)
        r = np.random.randint(0, 2, size=(n,)) * 2 - 1
        
        Br = B @ r      # O(n^2)
        C_r = C @ r     # O(n^2)
        ABr = A @ Br    # O(n^2)
        
        if not np.allclose(ABr, C_r, atol=1e-8):
            return False  # –¢–æ—á–Ω–æ –Ω–µ —Ä–∞–≤–Ω—ã
    return True           # –°–∫–æ—Ä–µ–µ –≤—Å–µ–≥–æ —Ä–∞–≤–Ω—ã

import numpy as np

def randomized_matrix_multiplication(A, B, k):
    """–ü—Ä–∏–±–ª–∏–∂–µ–Ω–Ω–æ–µ –º–∞—Ç—Ä–∏—á–Ω–æ–µ —É–º–Ω–æ–∂–µ–Ω–∏–µ AB —Å —Å—ç–º–ø–ª–∏—Ä–æ–≤–∞–Ω–∏–µ–º."""
    m, n = A.shape
    n_b, p = B.shape
    if n != n_b:
        raise ValueError("–ù–µ—Å–æ–≤–º–µ—Å—Ç–∏–º—ã–µ —Ä–∞–∑–º–µ—Ä—ã –º–∞—Ç—Ä–∏—Ü")

    # –í—ã—á–∏—Å–ª—è–µ–º –Ω–æ—Ä–º—ã –¥–ª—è –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–µ–π (–º–æ–∂–Ω–æ —É–ø—Ä–æ—Å—Ç–∏—Ç—å –¥–æ p_i = 1/n)
    col_norms_A = np.linalg.norm(A, axis=0)
    row_norms_B = np.linalg.norm(B, axis=1)
    probs = col_norms_A * row_norms_B
    sum_probs = np.sum(probs)
    if sum_probs == 0: # –ï—Å–ª–∏ –æ–¥–Ω–∞ –∏–∑ –º–∞—Ç—Ä–∏—Ü –Ω—É–ª–µ–≤–∞—è
         return np.zeros((m, p))
    probs /= sum_probs

    # –ú–æ–∂–Ω–æ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å —Ä–∞–≤–Ω–æ–º–µ—Ä–Ω–æ–µ —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ p_i = 1/n
    # probs = np.ones(n) / n

    C_approx = np.zeros((m, p))
    indices = np.random.choice(n, size=k, p=probs)
    # –ï—Å–ª–∏ –∏—Å–ø–æ–ª—å–∑—É–µ–º —Ä–∞–≤–Ω–æ–º–µ—Ä–Ω–æ–µ —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ:
    # indices = np.random.choice(n, size=k)
    # probs_selected = np.ones(k) / n

    probs_selected = probs[indices]

    for t in range(k):
        idx = indices[t]
        p_i = probs_selected[t]
        if p_i > 1e-9: # –ò–∑–±–µ–≥–∞–µ–º –¥–µ–ª–µ–Ω–∏—è –Ω–∞ –Ω–æ–ª—å
            A_col = A[:, idx:idx+1] # A^{(i_t)} –∫–∞–∫ —Å—Ç–æ–ª–±–µ—Ü (m x 1)
            B_row = B[idx:idx+1, :] # B_{(i_t)} –∫–∞–∫ —Å—Ç—Ä–æ–∫–∞ (1 x p)
            C_approx += (1.0 / (k * p_i)) * (A_col @ B_row)

    return C_approx

import numpy as np

def randomized_matmul(A, B, k):
    """
    –ü—Ä–∏–±–ª–∏–∂—ë–Ω–Ω–æ–µ —É–º–Ω–æ–∂–µ–Ω–∏–µ –º–∞—Ç—Ä–∏—Ü A (m x p) –∏ B (p x n).
    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç –º–∞—Ç—Ä–∏—Ü—É C ~ A @ B.
    """
    m, p = A.shape
    p2, n = B.shape
    assert p == p2
    
    # –í–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–∏ –±–µ—Ä—ë–º –ø—Ä–æ–ø–æ—Ä—Ü–∏–æ–Ω–∞–ª—å–Ω–æ –Ω–æ—Ä–º–∞–º —Å—Ç–æ–ª–±—Ü–æ–≤ A * —Å—Ç—Ä–æ–∫ B
    col_norms = np.linalg.norm(A, axis=0)  # —Ä–∞–∑–º–µ—Ä p
    row_norms = np.linalg.norm(B, axis=1)  # —Ä–∞–∑–º–µ—Ä p
    weights = col_norms * row_norms
    weights = weights / np.sum(weights)
    
    # –°—ç–º–ø–ª–∏–º k –∏–Ω–¥–µ–∫—Å–æ–≤ –±–µ–∑ –≤–æ–∑–≤—Ä–∞—â–µ–Ω–∏—è
    idx = np.random.choice(p, size=k, replace=False, p=weights)
    
    # –§–æ—Ä–º–∏—Ä—É–µ–º —Å—É–º–º—É
    C_approx = np.zeros((m, n))
    for i in idx:
        # A^{(i)} -- i-—ã–π —Å—Ç–æ–ª–±–µ—Ü A, B_{(i)} -- i-–∞—è —Å—Ç—Ä–æ–∫–∞ B
        # –£—á–∏—Ç—ã–≤–∞–µ–º –º–∞—Å—à—Ç–∞–±–∏—Ä–æ–≤–∞–Ω–∏–µ 1/(k * p_i)
        C_approx += (1.0 / (k * weights[i])) * np.outer(A[:, i], B[i, :])
    
    return C_approx

import numpy as np

def hutchinson_trace_estimator(A, k):
    """–û—Ü–µ–Ω–∫–∞ —Å–ª–µ–¥–∞ Tr(A) –º–µ—Ç–æ–¥–æ–º –•–∞—Ç—á–∏–Ω—Å–æ–Ω–∞."""
    n = A.shape[0]
    trace_estimates = []
    for _ in range(k):
        w = np.random.choice([-1.0, 1.0], size=(n, 1))
        trace_estimate = (w.T @ A @ w).item() # w^T A w
        trace_estimates.append(trace_estimate)
    return np.mean(trace_estimates)

def girard_trace_estimator(A, k):
    """–û—Ü–µ–Ω–∫–∞ —Å–ª–µ–¥–∞ Tr(A) –º–µ—Ç–æ–¥–æ–º –ñ–∏—Ä–∞—Ä–¥–∞."""
    n = A.shape[0]
    trace_estimates = []
    for _ in range(k):
        w = np.random.randn(n, 1) # w ~ N(0, I)
        trace_estimate = (w.T @ A @ w).item() # w^T A w
        trace_estimates.append(trace_estimate)
    return np.mean(trace_estimates)

    """

    -----------
    A_mv : function

    n : int
        –†–∞–∑–º–µ—Ä–Ω–æ—Å—Ç—å –º–∞—Ç—Ä–∏—Ü—ã A (n x n).
    k : int
        –ß–∏—Å–ª–æ —Å–ª—É—á–∞–π–Ω—ã—Ö –≤–µ–∫—Ç–æ—Ä–æ–≤ –¥–ª—è —É—Å—Ä–µ–¥–Ω–µ–Ω–∏—è.
    """
    estimates = []
    for _ in range(k):

        r = np.random.randint(0, 2, size=n)*2 - 1
        Ar = A_mv(r)
        estimates.append(r @ Ar)  # —Å–∫–∞–ª—è—Ä w^T (A w)
    return np.mean(estimates)

import numpy as np

def girard_trace_estimator(A_mv, n, k=10):
    """
    –û—Ü–µ–Ω–∏–≤–∞–µ—Ç Tr(A), –∏—Å–ø–æ–ª—å–∑—É—è –≥–∞—É—Å—Å–æ–≤—Å–∫–∏–µ —Å–ª—É—á–∞–π–Ω—ã–µ –≤–µ–∫—Ç–æ—Ä—ã.

    for _ in range(k):
        w = np.random.randn(n)  # –≥–∞—É—Å—Å–æ–≤—Å–∫–∏–π –≤–µ–∫—Ç–æ—Ä
        Aw = A_mv(w)
        estimates.append(w @ Aw)
    return np.mean(estimates)

def intrinsic_dimension(A):
    # –ü—Ä–µ–¥–ø–æ–ª–∞–≥–∞–µ–º, —á—Ç–æ A –∑–∞–¥–∞–Ω–∞ —è–≤–Ω–æ, —á—Ç–æ–±—ã –ø—Ä–æ—Å—Ç–æ –ø—Ä–æ–¥–µ–º–æ–Ω—Å—Ç—Ä–∏—Ä–æ–≤–∞—Ç—å
    trace_val = np.trace(A)
    fro_norm = np.linalg.norm(A, 'fro')
    return trace_val / fro_norm

import numpy as np

def randomized_svd(A, rank, oversampling=10):
    """–í—ã—á–∏—Å–ª—è–µ—Ç –ø—Ä–∏–±–ª–∏–∂–µ–Ω–Ω–æ–µ SVD —Ä–∞–Ω–≥–∞ 'rank'."""
    m, n = A.shape
    ell = rank + oversampling # –¶–µ–ª–µ–≤–æ–π —Ä–∞–Ω–≥ + –¥–æ–ø. —Å—ç–º–ø–ª–∏—Ä–æ–≤–∞–Ω–∏–µ

    # –®–∞–≥ 1: –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º —Å–ª—É—á–∞–π–Ω—É—é –º–∞—Ç—Ä–∏—Ü—É
    Omega = np.random.randn(n, ell)

    # –®–∞–≥ 2: –í—ã—á–∏—Å–ª—è–µ–º Y = A * Omega
    Y = A @ Omega

    # –®–∞–≥ 3: –ù–∞—Ö–æ–¥–∏–º –æ—Ä—Ç–æ–Ω–æ—Ä–º–∏—Ä–æ–≤–∞–Ω–Ω—ã–π –±–∞–∑–∏—Å Q –¥–ª—è Y (QR —Ä–∞–∑–ª–æ–∂–µ–Ω–∏–µ)
    Q, _ = np.linalg.qr(Y) # Q –∏–º–µ–µ—Ç —Ä–∞–∑–º–µ—Ä (m x ell)

    # –®–∞–≥ 4: –ü—Ä–æ–µ—Ü–∏—Ä—É–µ–º A –Ω–∞ –±–∞–∑–∏—Å Q: B = Q^T * A
    B = Q.T @ A # B –∏–º–µ–µ—Ç —Ä–∞–∑–º–µ—Ä (ell x n)

    # –®–∞–≥ 5: –í—ã—á–∏—Å–ª—è–µ–º SVD –¥–ª—è –º–∞–ª–µ–Ω—å–∫–æ–π –º–∞—Ç—Ä–∏—Ü—ã B
    U_tilde, s, Vh = np.linalg.svd(B, full_matrices=False) # U_tilde (ell x ell), s (ell,), Vh (ell x n)

    # –®–∞–≥ 6: –§–æ—Ä–º–∏—Ä—É–µ–º U –¥–ª—è –∏—Å—Ö–æ–¥–Ω–æ–π –º–∞—Ç—Ä–∏—Ü—ã A
    U = Q @ U_tilde # U –∏–º–µ–µ—Ç —Ä–∞–∑–º–µ—Ä (m x ell)

    # –®–∞–≥ 7: –û–±—Ä–µ–∑–∞–µ–º –¥–æ –Ω—É–∂–Ω–æ–≥–æ —Ä–∞–Ω–≥–∞ k
    U_k = U[:, :rank]
    s_k = s[:rank]
    Vh_k = Vh[:rank, :]

    return U_k, s_k, Vh_k

import numpy as np

def randomized_svd(A, k, p=5):
    """
    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç –ø—Ä–∏–±–ª–∏–∂–µ–Ω–∏–µ SVD –º–∞—Ç—Ä–∏—Ü—ã A: A ~ U @ np.diag(S) @ V.
    k - –∂–µ–ª–∞–µ–º—ã–π —Ä–∞–Ω–≥, p - –æ–≤–µ—Ä—Å—ç–º–ø–ª–∏–Ω–≥.
    """
    m, n = A.shape
    # 1. –°–ª—É—á–∞–π–Ω–∞—è –º–∞—Ç—Ä–∏—Ü–∞ G
    G = np.random.randn(n, k + p)
    # 2. Y = A G
    Y = A @ G  # —Ä–∞–∑–º–µ—Ä (m x (k+p))
    # 3. QR-—Ä–∞–∑–ª–æ–∂–µ–Ω–∏–µ
    Q, _ = np.linalg.qr(Y, mode='reduced')  # Q: (m x (k+p))
    # 4. B = Q^T A
    B = Q.T @ A  # —Ä–∞–∑–º–µ—Ä ((k+p) x n)
    # 5. –¢–æ—á–Ω–æ–µ SVD –º–∞—Ç—Ä–∏—Ü—ã B
    u_hat, s, v = np.linalg.svd(B, full_matrices=False)
    # 6. U = Q u_hat
    U = Q @ u_hat  # —Ä–∞–∑–º–µ—Ä (m x (k+p))
    # –û–±—Ä–µ–∂–µ–º –¥–æ —Ä–∞–Ω–≥–∞ k
    return U[:, :k], s[:k], v[:k, :]

import numpy as np

def randomized_kaczmarz(A, b, iters=1000):
    """
    –†–µ—à–∞–µ—Ç —Å–∏—Å—Ç–µ–º—É A x = b –º–µ—Ç–æ–¥–æ–º Kaczmarz.
    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç –ø—Ä–∏–±–ª–∏–∑–∏—Ç–µ–ª—å–Ω–æ–µ —Ä–µ—à–µ–Ω–∏–µ x.
    """
    m, n = A.shape
    x = np.zeros(n)
    row_norms = np.sum(A*A, axis=1)
    row_norms_sum = np.sum(row_norms)
    
    for _ in range(iters):
        # –í—ã–±–∏—Ä–∞–µ–º —Å—Ç—Ä–æ–∫—É i —Å–ª—É—á–∞–π–Ω–æ, –ø—Ä–æ–ø–æ—Ä—Ü. –µ–µ –Ω–æ—Ä–º–µ
        i = np.random.choice(m, p=row_norms/row_norms_sum)
        a_i = A[i, :]
        residual = a_i.dot(x) - b[i]
        x = x - (residual / (a_i.dot(a_i))) * a_i
    return x
